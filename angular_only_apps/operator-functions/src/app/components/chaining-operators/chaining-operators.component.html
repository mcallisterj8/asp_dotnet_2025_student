        tap((elem) => {
          this.tapInitalData.push(elem);
    
          /**
           * Uncomment the return to see 
           * that though no error will result
           * from returning from tap(), the 
           * value returned will always be
           * ignored downstream.
           */
          // return elem * 4;
        }),
        map((elem) => {    
          return elem * 2;
        }),
        filter((elem) => {        
          return elem % 2 == 0;
        })
      );
